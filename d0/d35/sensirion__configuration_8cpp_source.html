60 int8_t sensirion_i2c_read(uint8_t address, uint8_t* data, uint16_t count) {
61  uint8_t readData[count];
62  uint8_t rxByteCount = 0;
63 
64  // 2 bytes RH, 1 CRC, 2 bytes T, 1 CRC
65  Wire.requestFrom(address, count);
66 
67  while (Wire.available()) { // wait till all arrive
68  readData[rxByteCount++] = Wire.read();
69  if (rxByteCount >= count) {
70  break;
71  }
72  }
73 
74  memcpy(data, readData, count);
75 
76  return 0;
77 }
