96 bool Adafruit_BME280::init() {
97  // check if sensor, i.e. the chip ID is correct
98  _sensorID = read8(BME280_REGISTER_CHIPID);
99  if (_sensorID != 0x60)
100  return false;
101 
102  // reset the device using soft-reset
103  // this makes sure the IIR is off, etc.
104  write8(BME280_REGISTER_SOFTRESET, 0xB6);
105

263 bool Adafruit_BME280::takeForcedMeasurement(void) {
264  bool return_value = false;
265  // If we are in forced mode, the BME sensor goes back to sleep after each
266  // measurement and we need to set it to forced mode once at this point, so
267  // it will take the next measurement and then return to sleep again.
268  // In normal mode simply does new measurements periodically.
269  if (_measReg.mode == MODE_FORCED) {
270  return_value = true;
271  // set to forced mode, i.e. "take next measurement"
272  write8(BME280_REGISTER_CONTROL, _measReg.get());
273  // Store current time to measure the timeout
274  uint32_t timeout_start = millis();
275  // wait until measurement has been completed, otherwise we would read the
276  // the values from the last measurement or the timeout occurred after 2 sec.
277  while (read8(BME280_REGISTER_STATUS) & 0x08) {
278  // In case of a timeout, stop the while loop
279  if ((millis() - timeout_start) > 2000) {
280  return_value = false;
281  break;
282  }
283  delay(1);
284  }
285  }
286  return return_value;
287 }
